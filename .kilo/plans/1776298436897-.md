# LanceDB + AST-Aware Code Indexing System

## Objective
Build a high-performance local codebase indexing pipeline using LanceDB + Tree-sitter + ast-grep, supporting semantic + structural search for the gemini-web-wrapper project.

## Current State Analysis

The project already has a working repo indexing system (`repo_indexing.py`) that:
- Uses SQLite/libsql for storage
- Uses Tree-sitter + ast-grep for AST parsing
- Supports Python, Rust, Bash
- Has GitHub API integration for remote repos
- Has basic keyword-based search (no embeddings)

```python
"""Tree-sitter + ast-grep based AST parsing."""
from dataclasses import dataclass
from typing import Iterator
from tree_sitter import Node, Parser, Language
from ast_grep_py import SgRoot

@dataclass(frozen=True)
class ASTNode:
    """Represents a code unit extracted from AST."""
    path: str
    kind: str  # "function", "class", "method", "interface", etc.
    name: str
    start_byte: int
    end_byte: int
    start_line: int
    end_line: int
    code: str
    signature: str | None = None
    doc: str | None = None
    symbol_kind: str | None = None  # LSP symbol kind if available

@dataclass(frozen=True)
class PatternMatch:
    """ast-grep pattern match result."""
    path: str
    pattern: str
    kind: str = "pattern_match"
    name: str = ""
    start_byte: int = 0
    end_byte: int = 0
    start_line: int = 0
    end_line: int = 0
    code: str = ""

class LanguageConfig:
    """Configuration for a supported language."""
    
    def __init__(self, name: str, extensions: list[str], language: Language):
        self.name = name
        self.extensions = extensions
        self.language = language
        self.tree_sitter_kinds: dict[str, str] = {}  # node_type -> kind
        self.ast_grep_patterns: list[tuple[str, str, str]] = []  # (kind, pattern, capture)

class ASTParser:
    """Parse source files and extract AST nodes."""
    
    # Language configurations
    CONFIGS: dict[str, LanguageConfig] = {
        "python": LanguageConfig(
            name="python",
            extensions=[".py"],
            language=Language(tree_sitter_python.language()),
            tree_sitter_kinds={
                "function_definition": "function",
                "class_definition": "class",
                "method_definition": "method",
            },
            ast_grep_patterns=[
                ("function", "def $NAME($$$ARGS): $$$BODY", "NAME"),
                ("class", "class $NAME($$$ARGS): $$$BODY", "NAME"),
                ("class", "class $NAME: $$$BODY", "NAME"),
            ],
        ),
        "rust": LanguageConfig(...),
        "typescript": LanguageConfig(...),
        "javascript": LanguageConfig(...),
        "bash": LanguageConfig(...),
    }
    
    def __init__(self):
        self._parsers: dict[str, Parser] = {}
    
    def parse_file(self, path: str, content: str) -> Iterator[ASTNode]:
        """Parse file and yield all AST nodes."""
        language = self._detect_language(path)
        if not language:
            return
        
        # Tree-sitter extraction
        yield from self._extract_tree_sitter_nodes(path, content, language)
        
        # ast-grep pattern extraction
        yield from self._extract_pattern_matches(path, content, language)
    
    def _extract_tree_sitter_nodes(
        self, path: str, content: str, language: str
    ) -> Iterator[ASTNode]:
        """Extract nodes using Tree-sitter."""
        config = self.CONFIGS[language]
        parser = self._get_parser(language)
        tree = parser.parse(content.encode("utf-8"))
        
        def walk(node: Node) -> Iterator[ASTNode]:
            kind = config.tree_sitter_kinds.get(node.type)
            if kind:
                name = self._extract_node_name(node, content)
                if name:
                    yield ASTNode(
                        path=path,
                        kind=kind,
                        name=name,
                        start_byte=node.start_byte,
                        end_byte=node.end_byte,
                        start_line=node.start_point[0] + 1,
                        end_line=node.end_point[0] + 1,
                        code=content[node.start_byte:node.end_byte],
                    )
            for child in node.children:
                yield from walk(child)
        
        yield from walk(tree.root_node)
    
    def _extract_pattern_matches(
        self, path: str, content: str, language: str
    ) -> Iterator[PatternMatch]:
        """Extract nodes using ast-grep patterns."""
        config = self.CONFIGS.get(language)
        if not config or not config.ast_grep_patterns:
            return
        
        try:
            root = SgRoot(content, language)
            node = root.root()
        except Exception:
            return
        
        for kind, pattern, capture in config.ast_grep_patterns:
            try:
                matches = node.find_all(pattern=pattern)
                for match in matches:
                    name_node = match.get_match(capture)
                    if name_node is None:
                        continue
                    range_info = match.range()
                    yield PatternMatch(
                        path=path,
                        pattern=kind,
                        kind="pattern_match",
                        name=name_node.text(),
                        start_line=range_info.start.line + 1,
                        end_line=range_info.end.line + 1,
                        code=match.text(),
                    )
            except Exception:
                continue
```

#### 2.3 Chunker (`packages/code-index/src/affine/code_index/chunker.py`)

```python
"""Sliding window chunking for fallback recall."""
from dataclasses import dataclass

@dataclass(frozen=True)
class Chunk:
    """A text chunk for embedding."""
    path: str
    content: str
    start_byte: int
    end_byte: int
    start_line: int
    end_line: int
    index: int  # chunk sequence number

class SlidingWindowChunker:
    """Chunk files with overlap for semantic search fallback."""
    
    def __init__(self, chunk_size: int = 800, overlap: int = 200):
        self.chunk_size = chunk_size
        self.overlap = overlap
    
    def chunk_file(self, path: str, content: str) -> list[Chunk]:
        """Split file into overlapping chunks."""
        lines = content.split("\n")
        chunks = []
        start_line = 0
        chunk_index = 0
        
        while start_line < len(lines):
            # Build chunk up to chunk_size chars
            chunk_lines = []
            char_count = 0
            end_line = start_line
            
            while end_line < len(lines) and char_count + len(lines[end_line]) < self.chunk_size:
                chunk_lines.append(lines[end_line])
                char_count += len(lines[end_line]) + 1  # +1 for newline
                end_line += 1
            
            if not chunk_lines:
                # Single line exceeds chunk_size, take it anyway
                chunk_lines = [lines[start_line]]
                end_line = start_line + 1
            
            chunk_content = "\n".join(chunk_lines)
            
            # Calculate byte offsets
            start_byte = sum(len(l) + 1 for l in lines[:start_line])
            end_byte = start_byte + len(chunk_content)
            
            chunks.append(Chunk(
                path=path,
                content=chunk_content,
                start_byte=start_byte,
                end_byte=end_byte,
                start_line=start_line + 1,
                end_line=end_line,
                index=chunk_index,
            ))
            
            # Move window with overlap
            start_line = end_line - self.overlap // 50  # Approximate lines from chars
            if start_line <= chunk_index * (self.chunk_size - self.overlap) // 50:
                start_line = end_line
            chunk_index += 1
        
        return chunks
```

```python
"""LanceDB storage for code embeddings."""
import lancedb
import pyarrow as pa
import numpy as np
from pathlib import Path
from typing import Iterator
import hashlib
import json

class CodeIndexStore:
    """Manages LanceDB table for code embeddings."""
    
    TABLE_NAME = "code_index"
    
    SCHEMA = pa.schema([
        ("id", pa.string()),           # Unique ID: hash(path + kind + name + start_byte)
        ("path", pa.string()),         # File path
        ("kind", pa.string()),         # "function", "class", "chunk", "pattern_match"
        ("name", pa.string()),         # Symbol name or ""
        ("code", pa.string()),         # Full code snippet
        ("start_byte", pa.int64()),    # Byte offset start
        ("end_byte", pa.int64()),      # Byte offset end
        ("start_line", pa.int64()),    # Line number start
        ("end_line", pa.int64()),      # Line number end
        ("vector", pa.list_(pa.float32(), EMBEDDING_DIM)),  # Embedding vector
        ("pattern", pa.string()),      # ast-grep pattern name (if applicable)
        ("symbol_kind", pa.string()),  # LSP symbol kind (if available)
        ("doc", pa.string()),          # Docstring if extracted
        ("file_hash", pa.string()),    # Content hash for dedup
        ("indexed_at", pa.timestamp("us")),
    ])
    
    def __init__(self, db_path: Path, embedding_dim: int):
        self.db_path = db_path
        self.embedding_dim = embedding_dim
        self._db: lancedb.DBConnection | None = None
        self._table: lancedb.table.Table | None = None
    
    async def initialize(self) -> None:
        """Open/create database and table."""
        self.db_path.parent.mkdir(parents=True, exist_ok=True)
        self._db = await lancedb.connect_async(str(self.db_path))
        
        # Create table if not exists
        try:
            self._table = await self._db.open_table(self.TABLE_NAME)
        except Exception:
            # Create with schema
            schema = self._schema_with_dim()
            self._table = await self._db.create_table(
                self.TABLE_NAME,
                schema=schema,
                mode="create",
            )
    
    def _schema_with_dim(self) -> pa.Schema:
        """Build schema with correct embedding dimension."""
        fields = []
        for field in self.SCHEMA:
            if field.name == "vector":
                fields.append(("vector", pa.list_(pa.float32(), self.embedding_dim)))
            else:
                fields.append((field.name, field.type))
        return pa.schema(fields)
    
    async def upsert_batch(self, records: list[dict]) -> None:
        """Batch insert/update records."""
        if not records or self._table is None:
            return
        
        # Deduplicate by id
        seen = set()
        unique_records = []
        for record in records:
            if record["id"] not in seen:
                seen.add(record["id"])
                unique_records.append(record)
        
        await self._table.add(unique_records, mode="upsert")
    
    async def search(
        self,
        query_vector: list[float],
        k: int = 10,
        path_filter: str | None = None,
        kind_filter: str | None = None,
    ) -> list[dict]:
        """Semantic search with optional filters."""
        if self._table is None:
            raise RuntimeError("Store not initialized")
        
        builder = self._table.search(query_vector).metric("cosine")
        
        if path_filter:
            builder = builder.where(f"path LIKE '{path_filter}%'")
        if kind_filter:
            builder = builder.where(f"kind = '{kind_filter}'")
        
        results = await builder.limit(k).to_list()
        return results
    
    async def delete_by_file_hash(self, file_hash: str) -> None:
        """Remove all entries for a given file hash (for re-indexing)."""
        if self._table is None:
            return
        await self._table.delete(f"file_hash = '{file_hash}'")
    
    async def get_indexed_file_hashes(self) -> set[str]:
        """Get all file hashes currently in index."""
        if self._table is None:
            return set()
        
        # Query distinct file_hashes
        results = await self._table.query().select(["file_hash"]).distinct().to_list()
        return {r["file_hash"] for r in results}
```

#### 2.5 File Discovery (`packages/code-index/src/affine/code_index/discovery.py`)

```python
"""File discovery and change detection."""
from pathlib import Path
from typing import Iterator
import hashlib
import os

@dataclass
class FileInfo:
    path: str
    absolute_path: Path
    size: int
    mtime: float
    content_hash: str
    content: str

class FileDiscovery:
    """Discover and track source files."""
    
    DEFAULT_IGNORE = {
        ".git",
        "node_modules",
        "__pycache__",
        ".venv",
        "venv",
        ".cache",
        "dist",
        "build",
        ".index",  # Our own index dir
        "target",  # Rust build
    }
    
    DEFAULT_EXTENSIONS = {
        ".py", ".rs", ".ts", ".tsx", ".js", ".jsx", ".go",
        ".java", ".cpp", ".c", ".h", ".rb", ".php",
        ".swift", ".kt", ".scala", ".sh", ".bash",
    }
    
    def __init__(
        self,
        root: Path,
        extensions: set[str] | None = None,
        ignore_dirs: set[str] | None = None,
        max_file_bytes: int = 262_144,
    ):
        self.root = root.resolve()
        self.extensions = extensions or self.DEFAULT_EXTENSIONS
        self.ignore_dirs = ignore_dirs or self.DEFAULT_IGNORE
        self.max_file_bytes = max_file_bytes
    
    def discover(self) -> Iterator[FileInfo]:
        """Yield all matching files with content hashes."""
        for dirpath, dirnames, filenames in os.walk(self.root):
            # Filter ignored directories
            dirnames[:] = [
                d for d in dirnames
                if d not in self.ignore_dirs and not d.startswith(".")
            ]
            
            for filename in filenames:
                filepath = Path(dirpath) / filename
                
                # Check extension
                if filepath.suffix.lower() not in self.extensions:
                    continue
                
                # Check size
                try:
                    stat = filepath.stat()
                    if stat.st_size > self.max_file_bytes:
                        continue
                except OSError:
                    continue
                
                # Read and hash
                try:
                    content = filepath.read_text(encoding="utf-8", errors="ignore")
                    content_hash = hashlib.sha256(content.encode("utf-8")).hexdigest()[:16]
                except (OSError, UnicodeDecodeError):
                    continue
                
                yield FileInfo(
                    path=str(filepath.relative_to(self.root)),
                    absolute_path=filepath,
                    size=stat.st_size,
                    mtime=stat.st_mtime,
                    content_hash=content_hash,
                    content=content,
                )
    
    def compute_batch_hash(self, files: list[FileInfo]) -> str:
        """Compute aggregate hash for batch change detection."""
        hashes = sorted(f.content_hash for f in files)
        return hashlib.sha256("".join(hashes).encode()).hexdigest()[:16]
```

#### 3.1 Main Indexer (`packages/code-index/src/affine/code_index/indexer.py`)

```python
"""Main indexing orchestrator."""
import asyncio
from pathlib import Path
from datetime import datetime
from typing import Iterator
import hashlib

from .embedder import Embedder
from .parser import ASTParser, ASTNode, PatternMatch
from .chunker import SlidingWindowChunker, Chunk
from .store import CodeIndexStore
from .discovery import FileDiscovery, FileInfo

class CodeIndexer:
    """End-to-end code indexing pipeline."""
    
    def __init__(
        self,
        root: Path,
        embedder: Embedder,
        db_path: Path,
        chunk_size: int = 800,
        chunk_overlap: int = 200,
        batch_size: int = 100,
    ):
        self.root = root
        self.embedder = embedder
        self.db_path = db_path
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
        self.batch_size = batch_size
        
        self.parser = ASTParser()
        self.chunker = SlidingWindowChunker(chunk_size, chunk_overlap)
        self.discovery = FileDiscovery(root)
        self.store: CodeIndexStore | None = None
    
    async def initialize(self) -> None:
        """Initialize storage."""
        self.store = CodeIndexStore(self.db_path, self.embedder.dimension)
        await self.store.initialize()
    
    async def index(self, force: bool = False) -> dict:
        """Run full indexing pipeline."""
        if self.store is None:
            raise RuntimeError("Indexer not initialized")
        
        # Discover files
        files = list(self.discovery.discover())
        
        # Check for changes
        if not force:
            current_hashes = {f.content_hash for f in files}
            indexed_hashes = await self.store.get_indexed_file_hashes()
            
            if current_hashes == indexed_hashes:
                return {"status": "unchanged", "files": len(files), "indexed": 0}
        
        # Process in parallel batches
        stats = {"files": 0, "ast_nodes": 0, "chunks": 0, "errors": 0}
        
        for batch in self._batch(files, self.batch_size):
            batch_stats = await self._process_batch(batch)
            stats["files"] += batch_stats["files"]
            stats["ast_nodes"] += batch_stats["ast_nodes"]
            stats["chunks"] += batch_stats["chunks"]
            stats["errors"] += batch_stats["errors"]
        
        return {"status": "indexed", **stats}
    
    async def _process_batch(self, files: list[FileInfo]) -> dict:
        """Process a batch of files."""
        stats = {"files": 0, "ast_nodes": 0, "chunks": 0, "errors": 0}
        records = []
        
        for file_info in files:
            try:
                file_records = self._extract_records(file_info)
                records.extend(file_records)
                stats["files"] += 1
                stats["ast_nodes"] += len([r for r in file_records if r["kind"] != "chunk"])
                stats["chunks"] += len([r for r in file_records if r["kind"] == "chunk"])
            except Exception:
                stats["errors"] += 1
                continue
        
        # Batch embed
        if records:
            texts = [self._format_for_embedding(r) for r in records]
            vectors = await self.embedder.embed(texts)
            
            for record, vector in zip(records, vectors):
                record["vector"] = vector
            
            await self.store.upsert_batch(records)
        
        return stats
    
    def _extract_records(self, file_info: FileInfo) -> list[dict]:
        """Extract all indexable records from a file."""
        records = []
        seen_bytes: set[tuple[int, int]] = set()
        
        # 1. AST nodes (primary)
        ast_nodes = list(self.parser.parse_file(file_info.path, file_info.content))
        
        for node in ast_nodes:
            key = (node.start_byte, node.end_byte)
            if key in seen_bytes:
                continue
            seen_bytes.add(key)
            
            record_id = self._compute_id(file_info, node)
            records.append({
                "id": record_id,
                "path": node.path,
                "kind": node.kind if isinstance(node, ASTNode) else node.kind,
                "name": node.name,
                "code": node.code if isinstance(node, ASTNode) else node.code,
                "start_byte": node.start_byte,
                "end_byte": node.end_byte,
                "start_line": node.start_line,
                "end_line": node.end_line,
                "pattern": node.pattern if isinstance(node, PatternMatch) else None,
                "symbol_kind": node.symbol_kind if isinstance(node, ASTNode) else None,
                "doc": node.doc if isinstance(node, ASTNode) else None,
                "file_hash": file_info.content_hash,
                "indexed_at": datetime.now(),
            })
        
        # 2. Chunks (fallback for files with no AST nodes or for recall)
        if len(records) < 3:  # If few AST nodes, add chunks
            chunks = self.chunker.chunk_file(file_info.path, file_info.content)
            for chunk in chunks:
                # Skip chunks that overlap with AST nodes
                if any(
                    not (chunk.end_byte <= start or chunk.start_byte >= end)
                    for start, end in seen_bytes
                ):
                    continue
                
                record_id = self._compute_chunk_id(file_info, chunk)
                records.append({
                    "id": record_id,
                    "path": chunk.path,
                    "kind": "chunk",
                    "name": f"chunk_{chunk.index}",
                    "code": chunk.content,
                    "start_byte": chunk.start_byte,
                    "end_byte": chunk.end_byte,
                    "start_line": chunk.start_line,
                    "end_line": chunk.end_line,
                    "pattern": None,
                    "symbol_kind": None,
                    "doc": None,
                    "file_hash": file_info.content_hash,
                    "indexed_at": datetime.now(),
                })
        
        return records
    
    def _format_for_embedding(self, record: dict) -> str:
        """Format record for embedding."""
        parts = []
        if record["name"]:
            parts.append(f"{record['kind']}: {record['name']}")
        if record["doc"]:
            parts.append(record["doc"])
        parts.append(record["code"])
        return "\n".join(parts)
    
    def _compute_id(self, file_info: FileInfo, node: ASTNode | PatternMatch) -> str:
        """Compute unique ID for a node."""
        key = f"{file_info.path}:{node.kind}:{node.name}:{node.start_byte}"
        return hashlib.sha256(key.encode()).hexdigest()[:24]
    
    def _compute_chunk_id(self, file_info: FileInfo, chunk: Chunk) -> str:
        """Compute unique ID for a chunk."""
        key = f"{file_info.path}:chunk:{chunk.index}:{chunk.start_byte}"
        return hashlib.sha256(key.encode()).hexdigest()[:24]
    
    def _batch(self, items: list, size: int) -> Iterator[list]:
        """Batch items for processing."""
        for i in range(0, len(items), size):
            yield items[i:i + size]
```

#### 4.1 Search Engine (`packages/code-index/src/affine/code_index/search.py`)

```python
"""Hybrid semantic + structural search."""
from dataclasses import dataclass
from typing import Iterator

from .embedder import Embedder
from .store import CodeIndexStore

@dataclass
class SearchResult:
    """Search result with metadata."""
    path: str
    kind: str
    name: str
    code: str
    start_line: int
    end_line: int
    score: float
    is_ast_node: bool
    pattern: str | None = None

class CodeSearchEngine:
    """Hybrid search: AST nodes first, chunks as fallback."""
    
    def __init__(self, store: CodeIndexStore, embedder: Embedder):
        self.store = store
        self.embedder = embedder
    
    async def search(
        self,
        query: str,
        k: int = 10,
        path_prefix: str | None = None,
        prefer_ast: bool = True,
    ) -> list[SearchResult]:
        """Search with hybrid ranking."""
        # Embed query
        vectors = await self.embedder.embed([query])
        query_vector = vectors[0]
        
        # Search AST nodes first (higher precision)
        ast_results = await self.store.search(
            query_vector,
            k=k * 2 if prefer_ast else k,
            path_filter=path_prefix,
            kind_filter=None,  # Exclude chunks
        )
        
        # Filter to AST nodes only (exclude chunks)
```

```python
"""CLI for local codebase indexing."""
import asyncio
import sys
from pathlib import Path
import argparse

from affine.config.settings import get_settings
from affine.code_index.indexer import CodeIndexer
from affine.code_index.search import CodeSearchEngine
from affine.code_index.store import CodeIndexStore

async def index_command(args: argparse.Namespace) -> int:
    """Run indexing on local codebase."""
    settings = get_settings()
    
    # Select embedder based on settings
    if settings.model_provider == "gemini" and settings.google_api_key:
        embedder = GeminiEmbedder(settings.google_api_key)
    else:
        # Fallback to OpenAI-compatible
        embedder = OpenAIEmbedder(
            api_key=settings.provider_api_key() or "",
            base_url=settings.provider_base_url(),
        )
    
    root = Path(args.root or ".")
    db_path = Path(args.db or ".index/lancedb")
    
    indexer = CodeIndexer(

        embedder=embedder,
        db_path=db_path,
    )
    
    await indexer.initialize()
    result = await indexer.index(force=args.force)
    
    print(f"Indexing complete: {result}")
    return 0

async def search_command(args: argparse.Namespace) -> int:
    """Search the index."""
    settings = get_settings()
    
    # Same embedder selection as index
    if settings.model_provider == "gemini" and settings.google_api_key:
        embedder = GeminiEmbedder(settings.google_api_key)
    else:
        embedder = OpenAIEmbedder(
            api_key=settings.provider_api_key() or "",
            base_url=settings.provider_base_url(),
        )
    
    db_path = Path(args.db or ".index/lancedb")
    store = CodeIndexStore(db_path, embedder.dimension)
    await store.initialize()
    
    engine = CodeSearchEngine(store, embedder)
    results = await engine.search(
        query=args.query,
        k=args.k,
        path_prefix=args.path,
    )
    
    for i, result in enumerate(results, 1):
        print(f"\n{i}. {result.path}:{result.start_line}-{result.end_line}")
        print(f"   [{result.kind}] {result.name} (score: {result.score:.3f})")
        print(f"   {result.code[:200]}...")
    
    return 0

def main() -> int:
    """CLI entry point."""
    parser = argparse.ArgumentParser(description="Local codebase indexing")
    subparsers = parser.add_subparsers(dest="command")
    
    # Index command
    index_parser = subparsers.add_parser("index", help="Index codebase")
    index_parser.add_argument("--root", help="Root directory to index")
    index_parser.add_argument("--db", help="LanceDB path")
```

```python
"""FastAPI routes for local codebase indexing."""
from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel, Field

from affine.config.settings import Settings, get_settings
from affine.code_index.indexer import CodeIndexer
from affine.code_index.search import CodeSearchEngine
from affine.code_index.store import CodeIndexStore
from affine.code_index.embedder import OpenAIEmbedder, GeminiEmbedder

router = APIRouter(prefix="/v1/local-index", tags=["local-index"])

def get_embedder(settings: Settings):
    """Factory for embedder based on settings."""
    if settings.model_provider == "gemini" and settings.google_api_key:
        return GeminiEmbedder(settings.google_api_key)
    return OpenAIEmbedder(
        api_key=settings.provider_api_key() or "",
        base_url=settings.provider_base_url(),
    )

class LocalIndexRequest(BaseModel):
    root: str = Field(default=".", description="Root directory to index")
    force: bool = Field(default=False, description="Force re-index")

class LocalIndexResponse(BaseModel):
    status: str
    files: int
    ast_nodes: int
    chunks: int
    errors: int

class LocalSearchRequest(BaseModel):
    query: str = Field(min_length=1)
    k: int = Field(default=10, ge=1, le=50)
    path_prefix: str | None = None

class LocalSearchResult(BaseModel):
    path: str
    kind: str
    name: str
    code: str
    start_line: int
    end_line: int
    score: float
    is_ast_node: bool

class LocalSearchResponse(BaseModel):
    query: str
    results: list[LocalSearchResult]

@router.post("/index", response_model=LocalIndexResponse)
async def index_local(
    request: LocalIndexRequest,
    settings: Settings = Depends(get_settings),
):
    """Index local codebase."""
    embedder = get_embedder(settings)
    indexer = CodeIndexer(
        root=Path(request.root),
        embedder=embedder,
        db_path=settings.repo_index_db_path.parent / "lancedb",
    )
    await indexer.initialize()
    result = await indexer.index(force=request.force)
    
    return LocalIndexResponse(
        status=result["status"],
        files=result.get("files", 0),
        ast_nodes=result.get("ast_nodes", 0),
        chunks=result.get("chunks", 0),
        errors=result.get("errors", 0),
    )

@router.post("/search", response_model=LocalSearchResponse)
async def search_local(
    request: LocalSearchRequest,
    settings: Settings = Depends(get_settings),
):
    """Search local codebase index."""
    embedder = get_embedder(settings)
    store = CodeIndexStore(
        settings.repo_index_db_path.parent / "lancedb",
        embedder.dimension,
    )
    await store.initialize()
    
    engine = CodeSearchEngine(store, embedder)
    results = await engine.search(
        query=request.query,
        k=request.k,
        path_prefix=request.path_prefix,
    )
    
    return LocalSearchResponse(
        query=request.query,
        results=[
            LocalSearchResult(
                path=r.path,
                kind=r.kind,
                name=r.name,
                code=r.code[:500],  # Truncate for response
                start_line=r.start_line,
                end_line=r.end_line,
                score=r.score,
                is_ast_node=r.is_ast_node,
            )
            for r in results
        ],
    )
```
